### 求众数
```js
const find = array => {
   let count = 1
   let result = array[0]

   for (let i = 0; i < array.lenght; i++) {
       if (count === 0) result = array[i]

       if (array[i] === result) {
           count++
       }
       else {
           count--
       }
   }

   return result
}
```
### 洗牌算法
```js
const shuffle = function() {
   var array = this;
   var m = array.length,
       t, i;
   while (m) {
       i = Math.floor(Math.random() * m--);
       t = array[m];
       array[m] = array[i];
       array[i] = t;
   }
   return array;
}
```

### 斐波那契数列
```js
const m = new Map()
const mod = 1000000007
var fib = function(n) {
    if (n == 0) {
    return 0;
  } else if (n <= 2) { // if
    return 1;
  }
  if (m.has(n)) {
    return m.get(n);
  } else {
    let v = (fib(n - 1) + fib(n - 2)) % mod;
    m.set(n, v);
    return v;
  }
};
```
